1.npm init --yes

2.npm i express

3.expressUse.js：

//1.引入express
  const express = require('express');
//2.创建应用对象
  const app = express();
//3.创建路由规则
  app.get('/',(request,response)=>{
    response.send('HELLO,EXPRESS');
  });
//4.监听端口启动服务
  app.listen(8000,()=>{
      console.log("服务已经启动，8000端口监听中....")
  })
  
  *箭头函数需要ES6支持。
  
 4. 启动服务：
  node expressUse.js
 5. 结束服务：
  CTRL+C
