- 数组去重方法

```javascript
function arrayRepeat(arr){
	let result = [];
	for(let i=0;i<arr.length;i++){
		if(result.indexOf(arr[i]) == -1){
			result.push(arr[i]);
		}
	}
	
	return result;
}

let arr = [1,1,2,2,3,4];
console.log(arrayRepeat(arr));
```
